```python
# Python program to calculate the series and multiply by 4
total_sum = 0

for i in range(10000):
    term = 1 / (2 * i + 1)  # 1, 1/3, 1/5, ...
    if i % 2 == 0:
        total_sum += term  # Add for even index
    else:
        total_sum -= term  # Subtract for odd index

result = total_sum * 4
print(result)
```
